import dopamint as dp

# Initialize connection to physical agent
agent = dp.Agent(id="unit-07", secure=True)

# Subscribe to multi-modal sensory stream
@agent.on_perception
def handle_scene(scene: dp.SceneState):
    if scene.detect_anomaly():
        agent.halt()
        return dp.Action.EMERGENCY_STOP
        
    # Generate continuous control trajectory
    trajectory = dp.planner.calculate(scene)
    agent.execute(trajectory)
